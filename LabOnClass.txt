/* 多重join */
SELECT DISTINCT
    c.customerid,
    c.companyname,
    /*o.OrderID,*/
    od.productid,
    p.productname,
    sum(od.quantity)
FROM
    customers AS c
JOIN Orders AS o
	ON c.CustomerID = o.CustomerID
JOIN `order details` as od 
	on  o.OrderID = od.orderid
join products as p 
	on od.productid = p.productid  
WHERE c.CustomerID LIKE 'A%'
GROUP by c.CustomerID, c.CompanyName, p.ProductName , productid
ORDER BY `customerid` ,productid

index customers.CustomerID
      products.productid
      orders.OrderID
      order details.orderid

/* 關聯關係 */
create table customer(
    customerid int primary key,
    companyName varchar(20)
);
insert into customer values(1,'FW'),(2,"AHQ"),(3,'JT');

create table orders(
    orderId int primary key,
    customerid int
);
alter table orders  
    add constraint fk_orders_customer
    foreign key (customerid) references customer(customerid)
    on delete set null
    on update cascade;
/*
可以下的參數有：
[ON DELETE {CASCADE | SET NULL | NO ACTION | RESTRICT}]
[ON UPDATE {CASCADE | SET NULL | NO ACTION | RESTRICT}]

參數解釋：
    CASCADE - 會將有所關聯的紀錄行也會進行刪除或修改。
    SET NULL - 會將有所關聯的紀錄行設定成 NULL。
    NO ACTION - 有存在的關聯紀錄行時，會禁止父資料表的刪除或修改動作。
    RESTRICT - 與 NO ACTION 相同。
*/
